\[ \operatorname{I}=\int_0^{\pi/2}\frac{\left\{\cot x \right\}}{\cot x}\mathrm{dx} \] \[ \text{note}\ \left\{ \cot x\right\}=\frac{\cos x}{\sin x}-\left \lfloor \frac{\cos x}{\sin x}\right \rfloor \\ \begin{align*} \operatorname{I} &= \int_0^{\pi/2}\frac{\cot x}{\cot x} \mathrm{dx}-\int_0^{\pi/2}\frac{\left \lfloor \cot x\right \rfloor}{\cot x} \mathrm{dx} \\ &=\frac{\pi}{2}-\int_0^{\pi/2}\frac{\sin x}{\cos x}\left \lfloor \frac{\cos x}{\sin x} \right \rfloor \mathrm{dx}\\ &=\frac{\pi}{2}-\sum_{n\geq1}\int_{\operatorname{arccot} \left(n+1\right)}^{\operatorname{arccot} \left(n\right)}\left(n\tan \left(x\right)\right) \mathrm{dx} \\ &=\frac{\pi}{2}-\sum_{n\geq1}\left. \begin{matrix} \\ n\cdot\ln\left| \sec x \right| \\ \\ \end{matrix} \right|_{x=\operatorname{arccot} \left(n+1\right)}^{x=\operatorname{arccot} \left(n\right)} \\ &=\frac{\pi}{2}-\sum_{n\ge1}^{ }n\ln\left|\frac{\sqrt{n^{2}+1}}{n}\right|+\sum_{k\ge1}^{ }k\ln\left|\frac{\sqrt{\left(k+1\right)^{2}+1}}{k+1}\right| \\ &\text{let}\ k\rightarrow k-1 \\ &\Rightarrow\frac{\pi}{2}-\sum_{n\ge1}n\ln\left|\frac{\sqrt{n^{2}+1}}{n}\right|+\sum_{k\ge2}\left(k-1\right)\ln\left|\frac{\sqrt{k^{2}+1}}{k}\right| \\ &=\frac{\pi}{2}-\sum_{n\ge1}^{ }\ln\left(\frac{\sqrt{n^{2}+1}}{n}\right) \\ &=\frac{\pi}{2}-\frac{1}{2}\ln\left[\prod_{n\ge1}^{ }\left(1+\frac{1}{n^{2}}\right)\right] \\ &=\frac{\pi}{2}-\frac{1}{2}\ln\left(\frac{1}{\pi}\sinh\pi\right) \\ &\Rightarrow \operatorname{I}=\frac{1}{2}\ln\left(\frac{\pi\exp\pi}{\sinh\pi}\right) \end{align*} \]
